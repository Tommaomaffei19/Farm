local success, block = turtle.inspectDown()
function turtle.blockname()
  success, block = turtle.inspectDown()
  if block.name == nil then
   return "nothing here"
  else
   return block.name
  end
end
 
 turtle.up()
 turtle.forward()
 
for x = 1,21 do
if turtle.blockname() == "minecraft:pumpkin" then
 turtle.digDown()
 turtle.digDown()
 turtle.forward()
if turtle.blockname() == "minecraft:pumpkin_stem" then
  turtle.forward()
if turtle,blockname() == "minecraft:dirt" then
 turtle.digDown()
 turtle.forward()
if turtle.blockname() == "minecraft:grass" then
 turtle.digDown()
 turtle.forward()
 end
 end
 end
 end
end
